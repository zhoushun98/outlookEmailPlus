## 一、功能概述

### 1.1 设计目标

在**完全不改动后端代码**的前提下，对前端界面进行全面改造：
- 建立现代国风 CSS 设计系统（砖红/翠绿/琥珀金/宣纸米白）
- 用**固定三栏布局 + 左侧导航栏**替换现有四栏拖拽布局
- 所有 JS 业务函数**接口不变**（函数名/API URL/全局变量名不变），仅更新 DOM 结构适配

### 1.2 核心约束（优先级最高）

> ⚠️ **DOM ID 兼容性**：现有 JS 函数（在 index.html 内联脚本和 `static/js/features/*.js` 中）
> 通过大量 `document.getElementById(id)` 和 `querySelector` 操作 DOM。
> **所有被 JS 引用的 DOM element ID 必须保持不变**，或在对应 JS 文件中同步更新选择器。

关键保留 ID 清单：

### F1.2 深色模式切换

**功能编号**: F1.2  
**文件**: `static/css/main.css`（变量） + `templates/index.html`（JS）

#### 实现内容

```javascript
// index.html 内联初始化
function initTheme() {
  const saved = localStorage.getItem('ui_theme') || 'light';
  document.documentElement.dataset.theme = saved;
  updateThemeBtn(saved);
}
function toggleTheme() {
  const cur = document.documentElement.dataset.theme;
  const next = cur === 'dark' ? 'light' : 'dark';
  document.documentElement.dataset.theme = next;
  localStorage.setItem('ui_theme', next);
  updateThemeBtn(next);
}
function updateThemeBtn(theme) {
  const btn = document.getElementById('btnTheme');
  if (btn) btn.textContent = theme === 'dark' ? '☀ 浅色' : '☾ 深色';
}
```

**验收标准**:
- 切换后立即生效，无闪烁
- 刷新页面后主题持久化

### F2.2 导航激活状态

**功能编号**: F2.2

#### JS 实现

```javascript
function navigateTo(page) {
  // 更新 nav-item active
  document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
  const navMap = {
    'dashboard': 'navDashboard',
    'accounts':  'navAccounts',
    'tempmail':  'navTempmail',
    'groups':    'navGroups',
    'refresh-log': 'navRefreshLog',
    'audit':     'navAudit',
  };
  if (navMap[page]) {
    document.getElementById(navMap[page])?.classList.add('active');
  }
  // 显示对应内容区
  switchPage(page);
}
```

function updateSidebarBadges(groups) {

function stringToColor(str) {
  // 根据邮箱字符串稳定生成渐变色
  let hash = 0;
  for (let c of str) hash = c.charCodeAt(0) + ((hash << 5) - hash);
  const colors = [
    ['#4A90D9','#6BB3E0'], ['#E74C3C','#C0392B'], ['#27AE60','#1E8449'],
    ['#8E44AD','#6C3483'], ['#E67E22','#D35400'], ['#16A085','#1ABC9C']
  ];
  return `linear-gradient(135deg, ${colors[Math.abs(hash) % colors.length].join(',')})`;
}

function timeAgo(isoStr) {
  const diff = Date.now() - new Date(isoStr).getTime();
  const mins = Math.floor(diff / 60000);
  if (mins < 1) return '刚刚';
  if (mins < 60) return `${mins}分钟前`;
  const hrs = Math.floor(mins / 60);
  if (hrs < 24) return `${hrs}小时前`;
  return `${Math.floor(hrs/24)}天前`;
}

### F6.2 Token 状态徽章

**功能编号**: F6.2

```javascript
function getTokenStatus(account) {
  // account.token_expires_at 为 ISO 字符串，或 account.token_valid 为 bool
  if (!account.has_token) {
    return { badge: '<span class="badge badge-gray">— 无 Token</span>', timeClass: '' };
  }
  if (account.token_valid === false || account.token_status === 'expired') {
    return { badge: '<span class="badge badge-red">✗ 过期</span>', timeClass: 'text-danger' };
  }
  // 检查是否 7 天内过期
  if (account.token_expires_at) {
    const daysLeft = (new Date(account.token_expires_at) - Date.now()) / 86400000;
    if (daysLeft < 7) {
      return { badge: '<span class="badge badge-gold">⚠ 即将过期</span>', timeClass: 'text-warn' };
    }
  }
  return { badge: '<span class="badge badge-green">✓ 有效</span>', timeClass: '' };
}
```
